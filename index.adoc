=== Pre-Java 5: Fil I/O

[source,java]
----
import java.io.BufferedReader;
import java.io.FileReader;
import java.io.IOException;

public class ReadFileJava6 {
    public static void main(String[] args) {
        BufferedReader reader = null;
        try {
            reader = new BufferedReader(new FileReader("eksempel.txt"));
            String line;
            while ((line = reader.readLine()) != null) {
                System.out.println(line);
            }
        } catch (IOException e) {
            e.printStackTrace();
        } finally {
            if (reader != null) {
                try {
                    reader.close();
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
        }
    }
}

=== Dataklasser

[source,java,linenums]
----
class Ansat {
 
    private String fornavn;
    private String efternavn;
    private int Id;
 
    public Ansat(String fornavn, String efternavn,
                    int Id)
    {
 
        this.fornavn = fornavn;
        this.efternavn = efternavn;
        this.Id = Id;
    }
 
    public void setFornavn(String fornavn)
    {
        this.fornavn = fornavn;
    }
 
    public String getFornavn() { return fornavn; }
 
    public void setEfternavn(String efternavn)
    {
 
        this.efternavn = efternavn;
    }
 
    public String getEfternavn() { return efternavn; }
 
    public void setId(int Id) { this.Id = Id; }
 
    public int getId() { return Id; }
 
    public String toString()
    {
 
        return "Ansat [fornavn=" + fornavn
            + ", efternavn=" + efternavn + ", Id=" + Id + "]";
    }
 
    @Override public int hashCode()
    {
 
        final int prim = 31;
        int resultat = 1;
 
        resultat = prim * resultat + Id;
        resultat = prim * resultat
                 + ((fornavn == null)
                        ? 0
                        : fornavn.hashCode());
        resultat
            = prim * resultat
              + ((efternavn == null) ? 0
                                    : efternavn.hashCode());
 
        return resultat;
    }
 
    @Override public boolean equals(Object obj)
    {
 
        if (this == obj)
            return true;
 
        if (obj == null)
            return false;
 
        if (getClass() != obj.getClass())
            return false;
 
        Ansat anden = (Ansat)obj;
 
        if (Id != anden.Id)
            return false;
 
        if (fornavn == null) {
            if (anden.fornavn != null)
                return false;
        }
 
        else if (!fornavn.equals(anden.fornavn))
            return false;
 
        if (efternavn == null) {
            if (anden.efternavn != null)
                return false;
        }
 
        else if (!efternavn.equals(anden.efternavn))
            return false;
 
        return true;
    }
}
----
